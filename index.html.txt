            <button class="btn" onclick="showResult(1)">1- نخرج خروجة 🚗✨</button>
            <button class="btn" id="safariBtn" onmouseover="fleeButton()" onclick="handleSafariClick(event)">2- نسافر سفرية جامدة ✈️🌊</button>

    <script>
        // مصفوفة روابط لصور خرفان لطيفة ومبهجة
        const sheepImage = "https://img.icons8.com/color/96/cartoon-sheep.png";

        // توليد الخرفان الطائرة تلقائياً
        function createSheep() {
            const sheep = document.createElement('img');
            sheep.src = sheepImage;
            sheep.classList.add('sheep');
            
            // توزيع عشوائي للخرفان على الشاشة
            sheep.style.left = Math.random() * 100 + "vw";
            
            // أحجام وسرعات عشوائية لتأثير أجمل
            const size = Math.random() * 30 + 40; // بين 40px و 70px
            sheep.style.width = size + "px";
            
            const duration = Math.random() * 4 + 4; // بين 4 و 8 ثواني
            sheep.style.animationDuration = duration + "s";
            
            document.body.appendChild(sheep);
            
            // حذف الخروف بعد انتهاء حركته لتوفير الأداء
            setTimeout(() => {
                sheep.remove();
            }, duration * 1000);
        }

        // إطلاق خروف جديد كل نصف ثانية
        setInterval(createSheep, 500);

        let clickCount = 0;
        const safariBtn = document.getElementById('safariBtn');

        // وظيفة تجعل الزر يهرب عند اقتراب الماوس أو اللمس
        function fleeButton() {
            // الزر سيهرب فقط أول 3 مرات، في الرابعة سيثبت ليتم الضغط عليه وإظهار الصدمة
            if (clickCount < 3) {
                const x = Math.random() * (window.innerWidth - safariBtn.offsetWidth - 40) - (window.innerWidth / 2 - safariBtn.offsetWidth);
                const y = Math.random() * (window.innerHeight - safariBtn.offsetHeight - 40) - (window.innerHeight / 2 - safariBtn.offsetHeight);
                
                safariBtn.style.position = 'fixed';
                safariBtn.style.left = Math.min(Math.max(20, Math.random() * (window.innerWidth - 150)), window.innerWidth - 150) + 'px';
                safariBtn.style.top = Math.min(Math.max(20, Math.random() * (window.innerHeight - 80)), window.innerHeight - 80) + 'px';
            }
        }

        // معالجة الضغطات على زر السفرية
        function handleSafariClick(event) {
            event.preventDefault();
            clickCount++;

            if (clickCount === 1) {
                safariBtn.innerText = "متأكدة من الاختيار ده؟ 🤔";
                fleeButton();
            } else if (clickCount === 2) {
                safariBtn.innerText = "مفيش فايدة برضه؟ 😤";
                fleeButton();
            } else if (clickCount === 3) {
                safariBtn.innerText = "متأكدة آخر قرار؟ 🧐";
                // نلغي الهروب الآن لكي تستطيع ضغطه للمرة الرابعة والأخيرة
                safariBtn.style.position = 'relative';
                safariBtn.style.left = '0';
                safariBtn.style.top = '0';
            } else if (clickCount === 4) {
                showResult(2);
            }
        }

        // إظهار النتائج النهائية حسب الأرقام المطلوبة
        function showResult(choice) {
            const options = document.getElementById('options');
            const resultBox = document.getElementById('resultBox');
            
            options.style.display = 'none';
            
            let message = "";
            if(choice === 1) {
                message = "ماشي هنصرف 💸❤️";
                resultBox.style.color = "#2ed573";
            } else if(choice === 2) {
                message = "فوقي يا موهومة! 🤫😂";
                resultBox.style.color = "#ff4757";
            } else if(choice === 3) {
                message = "جدعة هيا دي نونو اللي أعرفها 🍿🛋️❤️";
                resultBox.style.color = "#1e90ff";
            }
            
            resultBox.innerHTML = message + "<br><br>🐑 كل سنة وأنتِ منورة حياتي يا روحي 🐑";
            resultBox.style.display = 'block';
        }
    </script>
